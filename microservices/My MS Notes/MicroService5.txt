message=Message from microservice-5

domain-c-service.ribbon.listOfServers=localhost:6125,localhost:6126
domain-c-service.ribbon.serverListRefreshInterval=5000


Annotations
------------
@EnableDiscoveryClient
@EnableHystrix
@EnableHystrixDashboard
@RibbonClient(name="domain-c-service")

@Bean
public Sampler defaultSampler() {
	return Sampler.ALWAYS_SAMPLE;
}

@LoadBalanced
@Bean
public RestTemplate getRestTemplate() {
	return new RestTemplate();
}
