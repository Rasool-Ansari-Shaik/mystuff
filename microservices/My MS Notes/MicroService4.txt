Annotations
------------
@EnableDiscoveryClient
@EnableHystrix
@EnableHystrixDashboard

@Bean
public Sampler defaultSampler() {
	return Sampler.ALWAYS_SAMPLE;
}

@LoadBalanced
@Bean
public RestTemplate getRestTemplate() {
	return new RestTemplate();
}
